```bash

uv init ai-chat-bot
cd ai-chat-bot
```

Initialize Git and create `.gitignore`:



## Step 7: Write the code

Create `src/ai_chat_bot/chat.py`:

```python
# src/ai_chat_bot/chat.py
from openai import OpenAI
from rich.console import Console
from rich.markdown import Markdown
import os

console = Console()

class ChatBot:
    def __init__(self):
        api_key = os.environ.get("OPENAI_API_KEY")
        if not api_key:
            raise ValueError("Please set OPENAI_API_KEY in .env file")
        
        self.client = OpenAI(api_key=api_key)
        self.messages = []
    
    def chat(self, user_input: str) -> str:
        self.messages.append({"role": "user", "content": user_input})
        
        response = self.client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=self.messages,
            temperature=0.7,
        )
        
        assistant_message = response.choices[0].message.content
        self.messages.append({"role": "assistant", "content": assistant_message})
        
        return assistant_message
```

Create `src/ai_chat_bot/main.py`:

```python
# src/ai_chat_bot/main.py
from dotenv import load_dotenv
from rich.console import Console
from rich.prompt import Prompt
from .chat import ChatBot

console = Console()

def main():
    # Load environment variables
    load_dotenv()
    
    # Create chatbot
    try:
        bot = ChatBot()
    except ValueError as e:
        console.print(f"[red]Error: {e}[/red]")
        return
    
    console.print("[bold green]AI Chat Bot[/bold green]")
    console.print("Type 'quit' to exit\n")
    
    while True:
        user_input = Prompt.ask("[blue]You[/blue]")
        
        if user_input.lower() == 'quit':
            break
        
        response = bot.chat(user_input)
        console.print(f"[green]Bot:[/green] {response}\n")
    
    console.print("[yellow]Goodbye![/yellow]")

if __name__ == "__main__":
    main()
```

## Step 10: Create tests

Create `tests/test_chat.py`:

```python
# tests/test_chat.py
import pytest
from unittest.mock import Mock, patch
from ai_chat_bot.chat import ChatBot

def test_chatbot_init_no_api_key():
    with patch.dict('os.environ', {}, clear=True):
        with pytest.raises(ValueError):
            ChatBot()

def test_chatbot_chat():
    with patch.dict('os.environ', {'OPENAI_API_KEY': 'test-key'}):
        with patch('ai_chat_bot.chat.OpenAI') as mock_openai:
            # Mock the API response
            mock_response = Mock()
            mock_response.choices = [Mock(message=Mock(content="Hello!"))]
            mock_openai.return_value.chat.completions.create.return_value = mock_response
            
            bot = ChatBot()
            response = bot.chat("Hi")
            
            assert response == "Hello!"
            assert len(bot.messages) == 2
```

Run tests:
```bash
uv run pytest
```
